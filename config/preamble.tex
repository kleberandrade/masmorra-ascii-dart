% Split de "Capítulo N — Título" em duas linhas (PDF)
% Pandoc pode gerar " --- " (três hífens) e quebras de linha; normalizamos antes de splitar.
\usepackage{xstring}
\newcommand{\ChapterTop}{}
\newcommand{\ChapterBottom}{}
\newcommand{\ChapterTitleNorm}{}
\newcommand{\SplitChapterTitle}[1]{%
  \StrSubstitute{#1}{ --- }{ — }[\ChapterTitleNorm]%
  \StrSubstitute{\ChapterTitleNorm}{  }{ }[\ChapterTitleNorm]%
  \expandafter\SplitChapterTitleAux\expandafter{\ChapterTitleNorm}%
}
\newcommand{\SplitChapterTitleAux}[1]{%
  \IfSubStr{#1}{ — }{%
    \StrBefore{#1}{ — }[\ChapterTop]%
    \StrBehind{#1}{ — }[\ChapterBottom]%
  }{%
    \def\ChapterTop{#1}%
    \def\ChapterBottom{}%
  }%
}

% Define Chapter format
% Sem ornamento e começando do topo.
\makeatletter
\def\@makechapterhead#1{%
  \vspace*{0\p@}%
  {\parindent \z@
   \normalfont\sffamily
   \SplitChapterTitle{#1}%
   \ifdefempty{\ChapterBottom}{%
     \filcenter\Large\bfseries\ChapterTop
   }{%
     \filcenter\small\textsc{\ChapterTop}%
     \\[8pt]\filcenter\LARGE\bfseries{\ChapterBottom}%
   }%
   \par\nobreak
   \vskip 24\p@
  }%
}
\def\@makeschapterhead#1{%
  \vspace*{0\p@}%
  {\parindent \z@
   \normalfont\sffamily
   \filcenter\Large\bfseries #1\par\nobreak
   \vskip 24\p@
  }%
}
\makeatother

% Start chapters on the next available page (avoid blank spreads)
% and keep the custom TOC entry formatting.
\usepackage{xparse}
\makeatletter
\@openrightfalse % Permite que os capitulos iniciem em qualquer pagina (mesmo par)
\pretocmd{\part}{\cleardoublepage}{}{} % Força que as Partes comecem em paginas impares
\let\stdchapter\chapter
\RenewDocumentCommand{\chapter}{s o m}{%
  \clearpage
  \IfBooleanTF{#1}{%
    \stdchapter*{#3}%
  }{%
    \SplitChapterTitle{#3}%
    \IfValueTF{#2}{%
       \stdchapter[#2]{#3}%
    }{%
       % Expand macros to avoid infinite recursion in \chaptermark.
       % Sem ":" quando não há subtítulo (ex.: Prefácio, Prólogo, Epílogo).
       \ifdefempty{\ChapterBottom}{%
         \protected@edef\TocEntry{\ChapterTop}%
       }{%
         \protected@edef\TocEntry{\ChapterTop: \ChapterBottom}%
       }%
       \stdchapter[\TocEntry]{#3}%
    }%
  }%
}
\makeatother

% Remove page numbers from blank pages

  \clearpage
  \if@twoside
    \ifodd\c@page
    \else
      \hbox{}\newpage
      \if@twocolumn
        \hbox{}\newpage
      \fi
    \fi
  \fi
}
\makeatother
